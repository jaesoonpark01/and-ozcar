동선 순서: 엔진룸 → 전면/외관 → 하체(리프트) → 실내 → 후면 → 최종 시운전. 마스터는 걷기만 하면 화면이 따라옵니다.

3. Supabase 기반 데이터 로직 (Backend Strategy)
사용자가 시뮬레이션을 돌릴 때마다 가상 시나리오를 연산하는 로직입니다.

SQL
-- 사용자의 정비 여부에 따른 미래 가치 시뮬레이션 함수
CREATE OR REPLACE FUNCTION simulate_future_value(
    p_vehicle_id UUID,
    p_action_taken BOOLEAN -- 정비 이행 여부
) RETURNS INT AS $$
DECLARE
    v_current_val INT;
    v_depreciation_rate FLOAT;
BEGIN
    SELECT current_market_value INTO v_current_val FROM car_assets WHERE id = p_vehicle_id;
    
    -- 정비를 안 하면 감가율 증가, 하면 방어
    IF p_action_taken THEN
        v_depreciation_rate := 0.05; -- 연 5% 감가
    ELSE
        v_depreciation_rate := 0.15; -- 연 15% 감가 (방치 시)
    END IF;

    RETURN v_current_val * (1 - v_depreciation_rate);

SELECT 
    ST_SnapToGrid(location, 0.001) as grid_point, -- 위치 정보를 그리드화(익명화)
    AVG(speed) as avg_speed,
    COUNT(*) filter (where sudden_brake = true) as brake_incidents,
    EXTRACT(HOUR FROM created_at) as hour_of_day,
    fuel_type
FROM raw_telemetry
GROUP BY grid_point, hour_of_day, fuel_type;
