// supabase/functions/create-stripe-session/index.ts
import { Stripe } from 'https://esm.sh/stripe@12.0.0'

const stripe = new Stripe(Deno.env.get('STRIPE_SECRET_KEY')!, { apiVersion: '2022-11-15' })

Deno.serve(async (req) => {
  const { user_id, price_id } = await req.json() // price_id는 Stripe 대시보드에서 생성

  const session = await stripe.checkout.sessions.create({
    customer_email: user_id, // 실제 유저 이메일 매핑
    payment_method_types: ['card'],
    line_items: [{ price: price_id, quantity: 1 }],
    mode: 'subscription',
    success_url: `https://ozcar.io/payment/success?session_id={CHECKOUT_SESSION_ID}`,
    cancel_url: `https://ozcar.io/payment/cancel`,
  })

  return new Response(JSON.stringify({ url: session.url }), { status: 200 })
})
