TypeScript
import { NextRequest, NextResponse } from 'next/server';
import { createClient } from '@supabase/supabase-js';

// Supabase 관리자 클라이언트 초기화
const supabase = createClient(
  process.env.SUPABASE_URL!,
  process.env.SUPABASE_SERVICE_ROLE_KEY!
);

export async function POST(req: NextRequest) {
  try {
    // 1. OBD 스캐너로부터 전달받은 원시 데이터 파싱
    const { vehicle_id, technician_id, sensor_data, dtc_codes } = await req.json();

    /* sensor_data 예시: 
       { rpm: 2500, speed: 60, coolant_temp: 90, voltage: 14.2, SOH: 85 }
    */

    // 2. AI Sentinel 로직: 이상 징후 즉시 판별 (Edge Computing)
    const is_critical = dtc_codes.length > 0 || sensor_data.coolant_temp > 110;

    // 3. Supabase 실시간 테이블에 데이터 삽입
    const { data, error } = await supabase
      .from('vehicle_realtime_logs')
      .insert({
        vehicle_id,
        technician_id,
        rpm: sensor_data.rpm,
        voltage: sensor_data.voltage,
        soh: sensor_data.SOH,
        coolant_temp: sensor_data.coolant_temp,
        dtc_codes: dtc_codes, // 고장코드 배열
        is_critical: is_critical,
        recorded_at: new Date().toISOString()
      })
      .select();

    if (error) throw error;

    // 4. 고장코드(DTC)가 있을 경우 별도의 경고 알림 프로세스 실행
    if (is_critical) {
      await triggerUrgentAlert(vehicle_id, dtc_codes);
    }

    return NextResponse.json({ success: true, log_id: data[0].id });

  } catch (error: any) {
    console.error('OBD Data Ingestion Error:', error.message);
    return NextResponse.json({ error: 'Data processing failed' }, { status: 500 });
  }
}

TypeScript
// services/ai/PredictiveEngine.ts

interface VehicleHealth {
  oil_viscosity_index: number; // 오일 점도 지수 (센서 추정치)
  avg_rpm: number;
  total_mileage: number;
  last_service_mileage: number;
  driving_style_score: number; // 1(온건) ~ 5(공격적)
}

export const calculateNextService = (data: VehicleHealth) => {
  // 1. 기본 수명 설정 (예: 엔진오일 10,000km)
  const BASE_LIFESPAN = 10000;
  const mileageSinceLastService = data.total_mileage - data.last_service_mileage;

  // 2. 가중치 계산 (운전 습관 및 실시간 상태 반영)
  // 공격적 운전은 소모품 수명을 더 빠르게 단축시킴
  const wearRate = (data.driving_style_score * 0.2) + (data.avg_rpm > 3000 ? 1.5 : 1.0);
  
  // 3. 잔여 수명 예측 (RUL)
  // $RUL = (Lifespan - CurrentUsage) / WearRate$
  const remainingLifeKm = Math.max(0, (BASE_LIFESPAN - mileageSinceLastService) / wearRate);
  
  // 4. 교체 시기 등급 분류
  let urgency: 'GOOD' | 'CAUTION' | 'URGENT' = 'GOOD';
  if (remainingLifeKm < 500) urgency = 'URGENT';
  else if (remainingLifeKm < 2000) urgency = 'CAUTION';

  return {
    remainingLifeKm: Math.floor(remainingLifeKm),
    urgency,
    predictedDate: calculateEstimatedDate(remainingLifeKm) // 주간 평균 주행거리 기반 날짜 계산
  };
};

function calculateEstimatedDate(km: number): Date {
  const avgDailyKm = 45; // 사용자의 일일 평균 주행 거리 데이터 (DB 연동 가능)
  const daysLeft = km / avgDailyKm;
  const today = new Date();
  today.setDate(today.getDate() + daysLeft);
  return today;
}

Data Freshness: 최근에 이루어진 검증된 정비일수록 감가 방어율을 높게 측정합니다.

// services/valuation/ValuationEngine.ts

export const calculateDynamicValue = (baseValue: number, repairLogs: any[]) => {
  // 1. AI 검증이 완료된(Status: APPROVED) 로그만 필터링
  const verifiedLogs = repairLogs.filter(log => log.ai_status === 'APPROVED');

  // 2. 각 정비 건별 가치 가산액 계산
  const totalPremium = verifiedLogs.reduce((acc, log) => {
    // 부품 중요도(Criticality)와 정비 품질(AI Score)에 따른 가산
    const partWeight = getPartWeight(log.part_type); // 예: 엔진 1.0, 소모품 0.2
    const qualityMultiplier = log.ai_confidence_score / 100;
    
    // 정비 후 경과 시간에 따른 가산액 감쇄 (최신 정비 우대)
    const decayFactor = calculateTimeDecay(log.repair_date);
    
    return acc + (log.cost * partWeight * qualityMultiplier * decayFactor);
  }, 0);

  return baseValue + totalPremium;
};

4. 기술적 구현: 자동 청구 및 스마트 클레임
(1) 구독 결제 (Stripe Billing)
Next.js API Route: 매월 1일, AI Sentinel이 지난달 주행 데이터(거리, 급제동 횟수)를 분석하여 discount_rate를 갱신하고, Stripe API를 통해 변동된 금액을 자동 결제합니다.

(2) 원클릭 수리 승인 (Smart Claim)
상황: 차주가 엔진 이상으로 ozcar 제휴 정비소에 입고.

AI 진단: 정비사가 OBD-II 스캐너를 연결하면, AI가 고장 코드를 분석하고 보증 범위 내 부품인지 즉시 판별.

자동 승인: 사람이 심사하지 않습니다. AI가 "정상적인 노후 파손"으로 판정하면, **ozcar 플랫폼이 정비사에게 수리비를 즉시 지급(OZC/Stablecoin)**하고 차주는 0원을 결제하고 출차합니다.

5. 비즈니스 확장 (Business Model)
Lock-in 효과: 구독 기간 동안에는 반드시 ozcar 제휴 정비소만 이용해야 하므로, 정비사들에게는 **고정적인 일감(Traffic)**을 제공하는 효과가 있습니다.

데이터 선순환: 구독자는 할인을 받기 위해 자발적으로 안전 운전을 하고 정비를 제때 받게 되며, 이는 ozcar에 양질의 데이터로 쌓입니다.

차주는 이제 '고장 걱정' 없이, 정비사는 '결제 걱정' 없이 오직 차에만 집중할 수 있습니다.
